<script>
  document.addEventListener('DOMContentLoaded', function () {
    document.getElementById('getMessage').onclick = function () {
      const req = new XMLHttpRequest();
      req.open('GET', '/json/cats.json', true);
      req.send();
      req.onload = function () {
        const json = JSON.parse(req.responseText);
        let html = '';
        // Add your code below this line
                // Loop through the JSON data
                json.forEach(function (val) {
                  const keys = Object.keys(val);
                  html += "<div class='cat'>"; // Start a new div for each cat

                  // Loop through each key-value pair in the object
                  keys.forEach(function (key) {
                    // Add key in <strong> and value in normal text
                    html += '<strong>' + key + ':</strong> ' + (Array.isArray(val[key]) ? val[key].join(", ") : val[key]) + '<br>';
                  });

                  html += '</div><br>'; // Close the cat div and add a line break
                });
                  
        // Add your code above this line
        document.getElementsByClassName('message')[0].innerHTML = html;
      };
    };
  });
</script>
