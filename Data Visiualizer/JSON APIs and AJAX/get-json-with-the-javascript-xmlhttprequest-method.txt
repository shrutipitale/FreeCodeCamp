<script>
  document.addEventListener('DOMContentLoaded', function () {
    document.getElementById('getMessage').onclick = function () {
      // Add your code below this line
       // 1. Create a new XMLHttpRequest
      const request = new XMLHttpRequest();

      // 2. Initialize a GET request to the freeCodeCamp Cat Photo API
      req.open("GET",'/json/cats.json' , true);



      // 4. Set an onload event handler
      request.onload = function () {
        // 5. Parse the responseText
        const data = JSON.parse(request.responseText);

        // 6. Change the innerHTML of the .message element to the string of JSON data
        document.getElementsByClassName('message')[0].innerHTML = JSON.stringify(data);
      };

      // 3. Send the request
      request.send();
      // Add your code above this line
    };
  });
</script>
